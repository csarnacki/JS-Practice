Creating a for loop.

let text = "";
for (i = 0; i < 6; i++) {
text += i + "
";
}


Creating a while loop.

let space = "";
let j = 10;
while(j <= 20) {
space += j + "
";
j++;
}


Looping through an array.

let blank = "";
let cars = ["BMW", "Volvo", "Jaguar", "Honda", "Mazda", "Ford", "Volkswagen"];
for(x = 0; x < cars.length; x++) {
blank += cars[x];
}


Breaking out of a loop.

let empty = "";
for(y = 0; y < 20; y++) {
if(y === 7) { break };
empty += y + "
";
}
